var name = $('#name').val(); insertWine(name, year, varietal, rating, description); function insertWine(name,year,varietal,rating,description) { db.transaction( function(transaction) { transaction.executeSql( 'INSERT INTO wine ' + ' (name, year, varietal, rating, description) VALUES ' + ' (?,?,?,?,?);',[name,year,varietal,rating,description] , successInsertHandler,errorHandler ); } ); } function successInsertHandler(transaction, error) { alert('Insert Success'); } function errorHandler(transaction, error) { alert('error ' + error.message + ' code: '+ error.code); } document.forms[0].reset();
